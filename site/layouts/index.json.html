{{/* Search index — outputs /index.json at build time. The digest markdown format puts the article title on one line: - **Title** `GA` and the description on the NEXT indented line: Description text here... [Link](url) We use a two-variable state machine: $category — current h2 section name $pendingTitle — an article title waiting for its description line */}} {{- $results := slice -}} {{- range (where .Site.RegularPages "Section" "posts").ByDate.Reverse -}} {{- $url := .RelPermalink -}} {{- $date := .Date.Format "2006-01-02" -}} {{- $category := "" -}} {{- $pendingTitle := "" -}} {{- range split .RawContent "\n" -}} {{- $line := . | strings.TrimRight "\r" -}} {{- $trimmed := strings.TrimSpace $line -}} {{/* ── h2 = new category section ── */}} {{- if hasPrefix $trimmed "## " -}} {{- if and $pendingTitle $category (not (strings.Contains ($category | lower) "top 5")) -}} {{- $results = $results | append (dict "title" $pendingTitle "category" $category "date" $date "url" $url "snippet" "") -}} {{- end -}} {{- $pendingTitle = "" -}} {{- $category = strings.TrimPrefix "## " $trimmed -}} {{/* ── h3 heading or bullet with bold title ── */}} {{- else if or (hasPrefix $trimmed "### ") (findRE `^\s*[-*\d]+\.?\s+\*\*` $trimmed) -}} {{/* Flush any previous pending title without a description */}} {{- if and $pendingTitle $category (not (strings.Contains ($category | lower) "top 5")) -}} {{- $results = $results | append (dict "title" $pendingTitle "category" $category "date" $date "url" $url "snippet" "") -}} {{- end -}} {{- $pendingTitle = "" -}} {{/* Only index non-Top-5 items */}} {{- if and $category (not (strings.Contains ($category | lower) "top 5")) -}} {{- if hasPrefix $trimmed "### " -}} {{- $pendingTitle = strings.TrimPrefix "### " $trimmed -}} {{- else -}} {{- $matches := findRE `\*\*(.+?)\*\*` $trimmed -}} {{- if $matches -}} {{- $pendingTitle = index $matches 0 | replaceRE `^\*\*|\*\*$` "" -}} {{- end -}} {{- end -}} {{- end -}} {{/* ── First real content line after a title = the description ── */}} {{- else if and $pendingTitle (not (eq $trimmed "")) -}} {{/* Skip link lines, hr lines, and phase-tag-only lines */}} {{- $isLink := hasPrefix $trimmed "[" -}} {{- $isSep := eq $trimmed "---" -}} {{- $isTagOnly := and (hasPrefix $trimmed "`") (le (len $trimmed) 30) -}} {{- if not (or $isLink $isSep $isTagOnly) -}} {{/* Strip leading backtick phase tags: `GA`, `GA — July CY2026`, etc. */}} {{- $snippet := replaceRE "^`[^`]+`\\s*[—–-]?\\s*" $trimmed "" | strings.TrimSpace -}} {{- if eq $snippet "" -}}{{- $snippet = $trimmed -}}{{- end -}} {{- if and $category (not (strings.Contains ($category | lower) "top 5")) -}} {{- $results = $results | append (dict "title" $pendingTitle "category" $category "date" $date "url" $url "snippet" $snippet) -}} {{- end -}} {{- $pendingTitle = "" -}} {{- end -}} {{- end -}} {{- end -}} {{/* Flush any trailing pending title at end of post */}} {{- if and $pendingTitle $category (not (strings.Contains ($category | lower) "top 5")) -}} {{- $results = $results | append (dict "title" $pendingTitle "category" $category "date" $date "url" $url "snippet" "") -}} {{- end -}} {{- end -}} {{- $results | jsonify -}}
